   166    mkTokenName base ix = fromString (Value.toString base ++ Haskell.show ix)

   168    {-# INLINABLE votingValue #-}
   169    votingValue :: MintingPolicyHash -> TokenName -> Value.Value
   170    votingValue mph tokenName =
   171        Value.singleton (Value.mpsSymbol mph) tokenName 1

   173    {-# INLINABLE ownsVotingToken #-}
   174    ownsVotingToken :: MintingPolicyHash -> TokenName -> TxConstraints Void Void
   175    ownsVotingToken mph tokenName = Constraints.mustSpendAtLeast (votingValue mph tokenName)

   177    {-# INLINABLE transition #-}
   178    transition :: Params -> State GovState -> GovInput -> Maybe (TxConstraints Void Void, State GovState)
   179    transition Params{..} State{ stateData = s, stateValue} i = case (s, i) of
   180    
   181        (GovState{mph}, MintTokens tokenNames) ->
   182            let (total, constraints) = foldMap
   183                    (\(pk, nm) -> let v = votingValue mph nm in (v, Constraints.mustPayToPubKey pk v))
   184                    (zip initialHolders tokenNames)
   185            in Just (constraints <> Constraints.mustMintValue total, State s stateValue)
   186    
   187        (GovState law mph Nothing, ProposeChange proposal@Proposal{tokenName}) ->
   188            let constraints = ownsVotingToken mph tokenName
   189            in Just (constraints, State (GovState law mph (Just (Voting proposal AssocMap.empty))) stateValue)
   190    
   191        (GovState law mph (Just (Voting p oldMap)), AddVote tokenName vote) ->
   192            let newMap = AssocMap.insert tokenName vote oldMap
   193                -- Correct validity interval should be:
   194                -- @
   195                --   Interval (LowerBound NegInf True) (Interval.strictUpperBound $ votingDeadline p)
   196                -- @
   197                -- See Note [Validity Interval's upper bound
   198                validityTimeRange = Interval.to (votingDeadline p - 2)
   199                constraints = ownsVotingToken mph tokenName
   200                            <> Constraints.mustValidateIn validityTimeRange
   201            in Just (constraints, State (GovState law mph (Just (Voting p newMap))) stateValue)
   202    
   203        (GovState oldLaw mph (Just (Voting p votes)), FinishVoting) ->
   204            let Sum ayes = foldMap (\b -> Sum $ if b then 1 else 0) votes
   205            in Just (mempty, State (GovState (if ayes >= requiredVotes then newLaw p else oldLaw) mph Nothing) stateValue)
   206    
   207        _ -> Nothing
